26/// An OpenDAL accessor that wraps another accessor and retries on PermissionDenied
27/// after refreshing credentials.
28///
29/// Each instance has its own inner accessor and shares credential state with
30/// other accessors via `Arc<RefreshableOpenDalStorage>`. Credentials are only
31/// refreshed when an operation fails with PermissionDenied, not proactively.
32///
33/// Concurrency: if multiple accessors hit PermissionDenied simultaneously, only
34/// one will call the external credential loader (via double-checked locking on
35/// `RefreshableOpenDalStorage::refresh_on_permission_denied`). The others will
36/// detect the version bump and simply rebuild their accessor from the already-refreshed
37/// credentials.
38pub(crate) struct RefreshableAccessor {
39    /// The current backend's accessor paired with the credential version it was built from.
40    inner: Mutex<(Accessor, u64)>,
41
42    /// The full original path (e.g. "memory:/some-file") used to create the operator.
43    /// Needed to rebuild the accessor after credential refresh.
44    original_path: String,
45
46    /// Shared storage holding credentials and configuration
47    storage: Arc<RefreshableOpenDalStorage>,
48}
49
50impl RefreshableAccessor {
51    pub(crate) fn new(
52        accessor: Accessor,
53        credential_version: u64,
54        original_path: String,
55        storage: Arc<RefreshableOpenDalStorage>,
56    ) -> Self {
57        Self {
58            inner: Mutex::new((accessor, credential_version)),
59            original_path,
60            storage,
61        }
62    }
63
64    /// Get the current inner accessor and its credential version.
65    fn get_accessor(&self) -> (Accessor, u64) {
66        let guard = self.inner.lock().unwrap();
67        guard.clone()
68    }
69
70    /// Rebuild the inner accessor from the shared storage after a credential refresh.
71    ///
72    /// Uses `original_path` (the full path passed to `refreshable_create_operator`)
73    /// to call `create_operator` on the refreshed `inner_storage`.
74    fn rebuild_accessor(&self, new_version: u64) -> Result<Accessor> {
75        let storage_guard = self.storage.lock_inner_storage();
76        let (operator, _) = storage_guard.create_operator(&self.original_path)?;
77        drop(storage_guard);
78
79        let new_accessor = operator.into_inner();
80        *self.inner.lock().unwrap() = (new_accessor.clone(), new_version);
81        Ok(new_accessor)
82    }
83
84    /// Run an operation with automatic retry on PermissionDenied after credential refresh.
85    ///
86    /// 1. Gets the current accessor (no refresh) and runs the operation.
87    /// 2. If it fails with PermissionDenied, calls `refresh_on_permission_denied`
88    ///    with the accessor's credential version.
89    /// 3. If credentials were refreshed (by us or another concurrent accessor),
90    ///    rebuilds our accessor and retries the operation once.
91    /// 4. Otherwise, returns the original error.
92    async fn with_credential_retry<F, Fut, T>(&self, op: F) -> opendal::Result<T>
93    where
94        F: Fn(Accessor) -> Fut,
95        Fut: Future<Output = opendal::Result<T>>,
96    {
97        let (accessor, version) = self.get_accessor();
98        let result = op(accessor).await;
99
100        match result {
101            Err(err) if err.kind() == opendal::ErrorKind::PermissionDenied => {
102                let new_version = self
103                    .storage
104                    .refresh_on_permission_denied(version)
105                    .await
106                    .map_err(|e| {
107                        opendal::Error::new(
108                            opendal::ErrorKind::PermissionDenied,
109                            format!(
110                                "Operation failed with PermissionDenied and credential \
111                                 refresh also failed: {e}"
112                            ),
113                        )
114                        .set_source(err)
115                    })?;
116
117                let new_accessor = self.rebuild_accessor(new_version).map_err(|e| {
118                    opendal::Error::new(
119                        opendal::ErrorKind::Unexpected,
120                        "Failed to rebuild accessor after credential refresh",
121                    )
122                    .set_source(e)
123                })?;
124                op(new_accessor).await
125            }
126            other => other,
127        }
128    }
129}

143    fn info(&self) -> Arc<AccessorInfo> {
144        let info_guard = self.storage.lock_cached_info();
145        if let Some(info) = info_guard.as_ref() {
146            Arc::clone(info)
147        } else {
148            drop(info_guard);
149            AccessorInfo::default().into()
150        }

159    }

207/// Tests for the `with_credential_retry` logic in `RefreshableAccessor`.
208///
209/// `with_credential_retry` works as follows:
210/// 1. Gets the current accessor (no refresh) and runs the operation.
211/// 2. On `PermissionDenied`, calls `refresh_on_permission_denied` with the
212///    accessor's credential version.
213/// 3. If credentials were refreshed, rebuilds the accessor and retries once.
214/// 4. Otherwise, returns the original error.
215///
216/// To test this, we inject a `FailingAccessor` (returns a configurable error on `stat`)
217/// as the initial inner accessor, while the shared storage's `inner_storage` is a real
218/// memory backend. When credential refresh triggers a rebuild, the accessor switches
219/// from `FailingAccessor` to the real memory backend — observable as a change in error
220/// kind (e.g. `PermissionDenied` → `NotFound`).
221///
222/// A `SequenceLoader` controls exactly which loader calls trigger refresh (`Some`) and
223/// which don't (`None`), so we can test each branch of the retry logic.
224#[cfg(all(test, feature = "storage-memory"))]
225mod tests {

235    /// Returns pre-configured credentials in order from a `VecDeque`. Tracks call count.
236    struct SequenceLoader {
237        responses: Mutex<VecDeque<StorageCredential>>,
238        call_count: AtomicUsize,
239    }
240
241    impl std::fmt::Debug for SequenceLoader {
242        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
243            f.debug_struct("SequenceLoader").finish()
244        }
245    }
246
247    impl SequenceLoader {
248        fn new(responses: Vec<StorageCredential>) -> Self {
249            Self {
250                responses: Mutex::new(VecDeque::from(responses)),
251                call_count: AtomicUsize::new(0),
252            }
253        }
254
255        fn call_count(&self) -> usize {
256            self.call_count.load(Ordering::SeqCst)
257        }
258    }
259
260    #[async_trait::async_trait]
261    impl StorageCredentialsLoader for SequenceLoader {
262        async fn load_credentials(&self, _location: &str) -> crate::Result<StorageCredential> {
263            self.call_count.fetch_add(1, Ordering::SeqCst);
264            let mut responses = self.responses.lock().unwrap();
265            Ok(responses.pop_front().unwrap_or_else(dummy_credential))
266        }
267    }

373    /// Builds a `RefreshableAccessor` whose initial inner accessor is a `FailingAccessor`
374    /// (returns `error_kind` on stat), but whose shared storage is a real memory backend.
375    /// After credential refresh + rebuild, the accessor switches from `FailingAccessor`
376    /// to the real memory backend.
377    fn build_refreshable_storage_and_accessor(
378        loader: Arc<dyn StorageCredentialsLoader>,
379        error_kind: opendal::ErrorKind,
380    ) -> RefreshableAccessor {
381        let storage = RefreshableOpenDalStorageBuilder::new()
382            .scheme("memory".to_string())
383            .base_props(HashMap::new())
384            .credentials_loader(Arc::clone(&loader))
385            .build()
386            .expect("Failed to build storage");
387
388        let info = {
389            let inner = storage.lock_inner_storage();
390            let path = "memory:/dummy".to_string();
391            let (op, _) = inner.create_operator(&path).unwrap();
392            op.into_inner().info()
393        };
394
395        *storage.lock_cached_info() = Some(Arc::clone(&info));
396
397        let version = storage.credential_version();
398        let failing_accessor: Accessor = Arc::new(FailingAccessor::new(error_kind, info));
399        RefreshableAccessor::new(
400            failing_accessor,
401            version,
402            "memory:/dummy".to_string(),
403            storage,
404        )
405    }

409    /// Core retry scenario: when temporary credentials expire mid-operation,
410    /// the accessor should transparently refresh and retry.
411    ///
412    /// Flow:
413    /// 1. `get_accessor` → no refresh → FailingAccessor used
414    /// 2. `stat` → PermissionDenied
415    /// 3. `refresh_on_permission_denied` → loader call #1 → do_refresh
416    /// 4. `rebuild_accessor` → memory accessor used
417    /// 5. Memory backend `stat("nonexistent")` → NotFound (not PermissionDenied)
418    #[tokio::test]
419    async fn test_retry_on_permission_denied_with_successful_refresh() {
420        let loader = Arc::new(SequenceLoader::new(vec![dummy_credential()]));
421
422        let accessor = build_refreshable_storage_and_accessor(
423            Arc::clone(&loader) as _,
424            opendal::ErrorKind::PermissionDenied,
425        );
426
427        let result = accessor.stat("nonexistent", OpStat::new()).await;
428
429        // The retry should have happened — the error should be NotFound
430        // (from the memory backend), not PermissionDenied
431        assert!(result.is_err());
432        let err = result.unwrap_err();
433        assert_eq!(
434            err.kind(),
435            opendal::ErrorKind::NotFound,
436            "Expected NotFound after retry, got {:?}",
437            err.kind()
438        );
439
440        // Only 1 loader call: the retry on PermissionDenied
441        assert_eq!(loader.call_count(), 1);
442    }

444    /// Only PermissionDenied should trigger credential retry. Other errors (network,
445    /// not-found, etc.) should not — retrying with fresh credentials wouldn't help.
446    ///
447    /// Flow:
448    /// 1. `get_accessor` → no refresh → FailingAccessor → NotFound
449    /// 2. `with_credential_retry` sees NotFound → no retry → returns error immediately
450    #[tokio::test]
451    async fn test_non_permission_denied_error_is_not_retried() {
452        let loader = Arc::new(SequenceLoader::new(vec![]));
453
454        let accessor = build_refreshable_storage_and_accessor(
455            Arc::clone(&loader) as _,
456            opendal::ErrorKind::NotFound,
457        );
458
459        let result = accessor.stat("nonexistent", OpStat::new()).await;
460
461        assert!(result.is_err());
462        let err = result.unwrap_err();
463        assert_eq!(err.kind(), opendal::ErrorKind::NotFound);
464
465        // No loader calls at all — only PermissionDenied triggers refresh
466        assert_eq!(loader.call_count(), 0);
467    }

469    /// When multiple concurrent callers hit PermissionDenied, only one should
470    /// call the external credential loader. The others should detect the version
471    /// bump and skip the loader call.
472    #[tokio::test]
473    async fn test_concurrent_permission_denied_calls_loader_only_once() {
474        let loader = Arc::new(SequenceLoader::new(vec![dummy_credential()]));
475
476        let storage = RefreshableOpenDalStorageBuilder::new()
477            .scheme("memory".to_string())
478            .base_props(HashMap::new())
479            .credentials_loader(Arc::clone(&loader) as _)
480            .build()
481            .expect("Failed to build storage");
482
483        let version = storage.credential_version();
484
485        // Spawn 10 concurrent refresh_on_permission_denied calls with the same version
486        let mut handles = Vec::new();
487        for _ in 0..10 {
488            let storage = Arc::clone(&storage);
489            handles.push(tokio::spawn(async move {
490                storage.refresh_on_permission_denied(version).await
491            }));
492        }
493
494        for handle in handles {
495            let new_version = handle.await.unwrap().unwrap();
496            assert_eq!(new_version, 1, "Version should be 1 after one refresh");
497        }
498
499        // Only 1 loader call should have been made
500        assert_eq!(loader.call_count(), 1);
501    }
