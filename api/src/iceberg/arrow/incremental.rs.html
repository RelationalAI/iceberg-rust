41/// Default batch size for incremental delete operations.
42const DEFAULT_BATCH_SIZE: usize = 1024;
43
44/// The type of incremental batch: appended data or deleted records.
45#[derive(Debug, Clone, Copy, PartialEq, Eq)]
46pub enum IncrementalBatchType {
47    /// Appended records.
48    Append,
49    /// Deleted records.
50    Delete,
51}
52
53/// Inner stream type for [`CombinedIncrementalScanResult`].
54pub type CombinedIncrementalBatchStream =
55    Pin<Box<dyn Stream<Item = Result<(IncrementalBatchType, RecordBatch)>> + Send + 'static>>;
56
57/// The stream of incremental Arrow `RecordBatch`es with batch type, together with scan metrics.
58pub struct CombinedIncrementalScanResult {
59    /// Combined stream of appended and deleted record batches, each tagged with its type.
60    pub stream: CombinedIncrementalBatchStream,
61    /// Metrics collected during the incremental scan (e.g. bytes read from storage).
62    pub metrics: ScanMetrics,
63}
64
65/// Separate streams for appended and deleted record batches, together with scan metrics.
66pub struct UnzippedIncrementalScanResult {
67    /// Stream of appended record batches.
68    pub appends: ArrowRecordBatchStream,
69    /// Stream of deleted record batches.
70    pub deletes: ArrowRecordBatchStream,
71    /// Metrics collected during the incremental scan (e.g. bytes read from storage).
72    pub metrics: ScanMetrics,
73}

75async fn process_incremental_append_task(
76    task: AppendedFileScanTask,
77    batch_size: Option<usize>,
78    file_io: FileIO,
79    parquet_read_options: ParquetReadOptions,
80    scan_metrics: ScanMetrics,
81) -> Result<ArrowRecordBatchStream> {
82    let AppendedFileScanTask {
83        base,
84        positional_deletes,
85        equality_delete_predicate,
86    } = task;
87
88    let equality_delete_bound = equality_delete_predicate
89        .map(|p| p.bind(base.schema.clone(), base.case_sensitive))
90        .transpose()?;
91
92    let (builder, has_missing_field_ids) = ArrowReader::open_parquet_stream_builder(
93        &base.data_file_path,
94        base.file_size_in_bytes,
95        file_io,
96        parquet_read_options,
97        ArrowReader::build_virtual_columns(&base.project_field_ids),
98        batch_size,
99        None, // name_mapping not yet supported in incremental scan
100        Some(Arc::clone(scan_metrics.bytes_read_counter())),
101        Some(&base.schema),
102    )
103    .await?;
104
105    let builder = ArrowReader::apply_parquet_filters(
106        builder,
107        base.start,
108        base.length,
109        &base.schema,
110        equality_delete_bound.as_ref(),
111        positional_deletes.as_deref(),
112        true,  // row_group_filtering_enabled
113        true,  // row_selection_enabled
114        false, // use_predicate_projection: projection applied separately via build_projected_record_batch_stream
115        has_missing_field_ids,
116    )?;
117
118    ArrowReader::build_projected_record_batch_stream(
119        builder,
120        &base.project_field_ids,
121        base.schema,
122        has_missing_field_ids,
123        &base.data_file_path,
124        base.partition_spec,
125        base.partition,
126    )
127}

129/// Helper function to create a RecordBatch from a chunk of position values.
130/// Creates a batch with file_path column first, then pos column (Int64).
131fn create_delete_batch(
132    schema: &Arc<ArrowSchema>,
133    file_path: &str,
134    chunk: Vec<u64>,
135) -> Result<RecordBatch> {
136    let num_rows = chunk.len();
137
138    // Create file path array (repeated for each row)
139    let file_array = arrow_array::StringArray::from(vec![file_path; num_rows]);
140
141    // Create Int64 array for positions
142    let pos_array = Int64Array::from_iter(chunk.iter().map(|&i| Some(i as i64)));
143
144    RecordBatch::try_new(Arc::clone(schema), vec![
145        Arc::new(file_array),
146        Arc::new(pos_array),
147    ])
148    .map_err(|_| {
149        Error::new(
150            ErrorKind::Unexpected,
151            "Failed to create RecordBatch for delete positions",
152        )
153    })
154}

156fn process_incremental_delete_task(
157    file_path: String,
158    delete_vector: DeleteVector,
159    batch_size: Option<usize>,
160) -> Result<ArrowRecordBatchStream> {
161    let batch_size = batch_size.unwrap_or(DEFAULT_BATCH_SIZE);
162
163    // Create schema with file_path column first, then pos (Int64)
164    let schema = Arc::new(ArrowSchema::new(vec![
165        Arc::clone(crate::metadata_columns::file_path_field()),
166        Arc::clone(crate::metadata_columns::pos_field_arrow()),
167    ]));
168
169    let treemap = delete_vector.inner;
170
171    let stream = futures::stream::iter(treemap)
172        .chunks(batch_size)
173        .map(move |chunk| create_delete_batch(&schema, &file_path, chunk));
174
175    Ok(Box::pin(stream) as ArrowRecordBatchStream)
176}

178fn process_incremental_deleted_file_task(
179    file_path: String,
180    total_records: u64,
181    batch_size: Option<usize>,
182) -> Result<ArrowRecordBatchStream> {
183    let batch_size = batch_size.unwrap_or(DEFAULT_BATCH_SIZE);
184
185    // Create schema with file_path column first, then pos (Int64)
186    let schema = Arc::new(ArrowSchema::new(vec![
187        Arc::clone(crate::metadata_columns::file_path_field()),
188        Arc::clone(crate::metadata_columns::pos_field_arrow()),
189    ]));
190
191    // Create a stream of position values from 0 to total_records-1 (0-indexed)
192    let stream = futures::stream::iter(0..total_records)
193        .chunks(batch_size)
194        .map(move |chunk| create_delete_batch(&schema, &file_path, chunk));
195
196    Ok(Box::pin(stream) as ArrowRecordBatchStream)
197}

199/// Process equality delete task by reading the data file with equality delete predicates applied
200/// as a row filter, and emitting record batches containing matching row positions.
201async fn process_equality_delete_task(
202    task: EqualityDeleteScanTask,
203    batch_size: Option<usize>,
204    file_io: FileIO,
205    parquet_read_options: ParquetReadOptions,
206    scan_metrics: ScanMetrics,
207) -> Result<ArrowRecordBatchStream> {
208    let file_path = task.data_file_path().to_string();
209
210    // Create output schema with file_path column first, then pos (Int64)
211    let output_schema = Arc::new(ArrowSchema::new(vec![
212        Arc::clone(crate::metadata_columns::file_path_field()),
213        Arc::clone(crate::metadata_columns::pos_field_arrow()),
214    ]));
215
216    let bound_predicate = task
217        .combined_predicate
218        .bind(task.schema_ref(), task.base.case_sensitive)?;
219
220    let (builder, has_missing_field_ids) = ArrowReader::open_parquet_stream_builder(
221        &task.base.data_file_path,
222        task.base.file_size_in_bytes,
223        file_io,
224        parquet_read_options,
225        vec![Arc::clone(row_pos_field())],
226        batch_size,
227        None, // name_mapping not yet supported in incremental scan
228        Some(Arc::clone(scan_metrics.bytes_read_counter())),
229        Some(&task.base.schema),
230    )
231    .await?;
232
233    let builder = ArrowReader::apply_parquet_filters(
234        builder,
235        task.base.start,
236        task.base.length,
237        &task.base.schema,
238        Some(&bound_predicate),
239        None,  // no positional deletes for equality delete tasks
240        true,  // row_group_filtering_enabled
241        false, // row_selection_enabled
242        true,  // use_predicate_projection: project to predicate columns only
243        has_missing_field_ids,
244    )?;
245
246    // Build the stream of filtered records
247    let record_batch_stream = builder.build()?;
248
249    // Extract positions from the _pos column and emit delete batches
250    let output_schema_clone = output_schema.clone();
251    let file_path_clone = file_path.clone();
252
253    let stream = record_batch_stream
254        .then(move |batch_result| {
255            let schema = output_schema_clone.clone();
256            let path = file_path_clone.clone();
257            async move {
258                match batch_result {
259                    Ok(batch) => {
260                        // Extract _pos column (last column due to virtual_columns).
261                        // _pos is always non-null: it is a virtual column representing the
262                        // physical row position, produced by the Parquet reader for every row.
263                        let pos_col = batch
264                            .column(batch.num_columns() - 1)
265                            .as_any()
266                            .downcast_ref::<Int64Array>()
267                            .ok_or_else(|| {
268                                Error::new(
269                                    ErrorKind::Unexpected,
270                                    "Failed to extract _pos column from equality delete batch",
271                                )
272                            })?;
273
274                        let num_rows = pos_col.len();
275                        if num_rows == 0 {
276                            return Ok(RecordBatch::new_empty(Arc::clone(&schema)));
277                        }
278
279                        // Reuse the Int64Array directly as the pos column.
280                        // Build a matching file_path StringArray.
281                        let file_array = Arc::new(arrow_array::StringArray::from(vec![
282                            path.as_str(
283                            );
284                            num_rows
285                        ]));
286                        RecordBatch::try_new(Arc::clone(&schema), vec![
287                            file_array,
288                            Arc::clone(batch.column(batch.num_columns() - 1)),
289                        ])
290                        .map_err(|e| {
291                            Error::new(
292                                ErrorKind::Unexpected,
293                                format!("Failed to create equality delete batch: {e}"),
294                            )
295                        })
296                    }
297                    Err(e) => Err(e.into()),
298                }
299            }
300        })
301        .boxed();
302
303    Ok(Box::pin(stream) as ArrowRecordBatchStream)
304}

306impl StreamsInto<ArrowReader, CombinedIncrementalScanResult> for IncrementalFileScanTaskStreams {
307    /// Takes separate streams of appended and deleted file scan tasks and reads all the files.
308    /// Returns a [`CombinedIncrementalScanResult`] containing a combined stream of Arrow
309    /// `RecordBatch`es and scan metrics.
310    fn stream(self, reader: ArrowReader) -> Result<CombinedIncrementalScanResult> {
311        let UnzippedIncrementalScanResult {
312            appends,
313            deletes,
314            metrics,
315        } = StreamsInto::<ArrowReader, UnzippedIncrementalScanResult>::stream(self, reader)?;
316
317        let left = appends.map(|res| res.map(|batch| (IncrementalBatchType::Append, batch)));
318        let right = deletes.map(|res| res.map(|batch| (IncrementalBatchType::Delete, batch)));
319
320        Ok(CombinedIncrementalScanResult {
321            stream: Box::pin(select(left, right)),
322            metrics,
323        })
324    }
325}

327impl StreamsInto<ArrowReader, UnzippedIncrementalScanResult> for IncrementalFileScanTaskStreams {
328    /// Takes separate streams of appended and deleted file scan tasks and reads all the files.
329    /// Returns an [`UnzippedIncrementalScanResult`] containing separate streams of appended and
330    /// deleted record batches together with scan metrics.
331    fn stream(self, reader: ArrowReader) -> Result<UnzippedIncrementalScanResult> {
332        let (appends_tx, appends_rx) =
333            channel::<Result<RecordBatch>>(reader.concurrency_limit_data_files);
334        let (deletes_tx, deletes_rx) =
335            channel::<Result<RecordBatch>>(reader.concurrency_limit_data_files);
336
337        let batch_size = reader.batch_size;
338        let parquet_read_options = reader.parquet_read_options;
339        let scan_metrics = ScanMetrics::new();
340
341        let (append_stream, delete_stream) = self;
342
343        // Process append tasks
344        let file_io_append = reader.file_io.clone();
345        let scan_metrics_append = scan_metrics.clone();
346        spawn(async move {
347            let _ = append_stream
348                .try_for_each_concurrent(reader.concurrency_limit_data_files, |append_task| {
349                    let file_io = file_io_append.clone();
350                    let appends_tx = appends_tx.clone();
351                    let scan_metrics = scan_metrics_append.clone();
352                    async move {
353                        // Inner spawn: each file's IO runs on its own tokio task for true
354                        // parallelism. Awaiting it keeps the concurrency slot occupied until
355                        // the file is fully read, matching the full-scan reader pattern.
356                        spawn(async move {
357                            let should_load_page_index =
358                                append_task.equality_delete_predicate.is_some()
359                                    || append_task.positional_deletes.is_some();
360                            let mut append_read_options = parquet_read_options;
361                            append_read_options.preload_page_index = should_load_page_index;
362                            let record_batch_stream = process_incremental_append_task(
363                                append_task,
364                                batch_size,
365                                file_io,
366                                append_read_options,
367                                scan_metrics,
368                            )
369                            .await;
370
371                            process_record_batch_stream(
372                                record_batch_stream,
373                                appends_tx,
374                                "failed to read appended record batch",
375                            )
376                            .await;
377                        })
378                        .await;
379
380                        Ok(())
381                    }
382                })
383                .await;
384        });
385
386        // Process delete tasks
387        let file_io_delete = reader.file_io.clone();
388        let scan_metrics_delete = scan_metrics.clone();
389        spawn(async move {
390            let _ = delete_stream
391                .try_for_each_concurrent(reader.concurrency_limit_data_files, |delete_task| {
392                    let deletes_tx = deletes_tx.clone();
393                    let file_io = file_io_delete.clone();
394                    let scan_metrics = scan_metrics_delete.clone();
395                    async move {
396                        // Inner spawn: same pattern as full-scan reader — spawn for parallelism,
397                        // await to keep the concurrency slot occupied until the task completes.
398                        spawn(async move {
399                            match delete_task {
400                                DeleteScanTask::DeletedFile(deleted_file_task) => {
401                                    let file_path = deleted_file_task.data_file_path().to_string();
402                                    let total_records =
403                                        deleted_file_task.base.record_count.unwrap_or(0);
404
405                                    let record_batch_stream = process_incremental_deleted_file_task(
406                                        file_path,
407                                        total_records,
408                                        batch_size,
409                                    );
410
411                                    process_record_batch_stream(
412                                        record_batch_stream,
413                                        deletes_tx,
414                                        "failed to read deleted file record batch",
415                                    )
416                                    .await;
417                                }
418                                DeleteScanTask::PositionalDeletes(file_path, delete_vector) => {
419                                    let record_batch_stream = process_incremental_delete_task(
420                                        file_path,
421                                        delete_vector,
422                                        batch_size,
423                                    );
424
425                                    process_record_batch_stream(
426                                        record_batch_stream,
427                                        deletes_tx,
428                                        "failed to read deleted record batch",
429                                    )
430                                    .await;
431                                }
432                                DeleteScanTask::EqualityDeletes(equality_delete_task) => {
433                                    // equality delete tasks always need the page index: we always have a predicate
434                                    let mut eq_read_options = parquet_read_options;
435                                    eq_read_options.preload_page_index = true;
436                                    let record_batch_stream = process_equality_delete_task(
437                                        equality_delete_task,
438                                        batch_size,
439                                        file_io.clone(),
440                                        eq_read_options,
441                                        scan_metrics,
442                                    )
443                                    .await;
444
445                                    process_record_batch_stream(
446                                        record_batch_stream,
447                                        deletes_tx,
448                                        "failed to read equality delete record batch",
449                                    )
450                                    .await;
451                                }
452                            }
453                        })
454                        .await;
455
456                        Ok(())
457                    }
458                })
459                .await;
460        });
461
462        Ok(UnzippedIncrementalScanResult {
463            appends: Box::pin(appends_rx) as ArrowRecordBatchStream,
464            deletes: Box::pin(deletes_rx) as ArrowRecordBatchStream,
465            metrics: scan_metrics,
466        })
467    }
