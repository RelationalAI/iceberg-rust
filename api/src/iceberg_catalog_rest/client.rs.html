32/// Trait for custom token authentication.
33///
34/// Implement this trait to provide custom token generation/refresh logic
35/// instead of using OAuth credentials.
36#[async_trait::async_trait]
37pub trait CustomAuthenticator: Send + Sync + Debug {
38    /// Get or refresh the authentication token.
39    /// Called when the client needs a token for authentication.
40    async fn get_token(&self) -> Result<String>;
41}
42
43pub(crate) struct HttpClient {
44    client: Client,
45
46    /// The token to be used for authentication.
47    ///
48    /// It's possible to fetch the token from the server while needed.
49    token: Mutex<Option<String>>,
50    /// The token endpoint to be used for authentication.
51    token_endpoint: String,
52    /// The credential to be used for authentication.
53    credential: Option<(Option<String>, String)>,
54    /// Custom token authenticator (takes precedence over credential/token)
55    authenticator: Option<Arc<dyn CustomAuthenticator>>,
56    /// Extra headers to be added to each request.
57    extra_headers: HeaderMap,
58    /// Extra oauth parameters to be added to each authentication request.
59    extra_oauth_params: HashMap<String, String>,
60}
61
62impl Debug for HttpClient {
63    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
64        f.debug_struct("HttpClient")
65            .field("client", &self.client)
66            .field("extra_headers", &self.extra_headers)
67            .finish_non_exhaustive()
68    }
69}

71impl HttpClient {
72    /// Create a new http client.
73    pub fn new(cfg: &RestCatalogConfig) -> Result<Self> {
74        let extra_headers = cfg.extra_headers()?;
75        Ok(HttpClient {
76            client: cfg.client().unwrap_or_default(),
77            token: Mutex::new(cfg.token()),
78            token_endpoint: cfg.get_token_endpoint(),
79            credential: cfg.credential(),
80            authenticator: None,
81            extra_headers,
82            extra_oauth_params: cfg.extra_oauth_params(),
83        })
84    }
85
86    /// Update the http client with new configuration.
87    ///
88    /// If cfg carries new value, we will use cfg instead.
89    /// Otherwise, we will keep the old value.
90    pub fn update_with(self, cfg: &RestCatalogConfig) -> Result<Self> {
91        let extra_headers = (!cfg.extra_headers()?.is_empty())
92            .then(|| cfg.extra_headers())
93            .transpose()?
94            .unwrap_or(self.extra_headers);
95        Ok(HttpClient {
96            client: cfg.client().unwrap_or(self.client),
97            token: Mutex::new(cfg.token().or_else(|| self.token.into_inner())),
98            token_endpoint: if !cfg.get_token_endpoint().is_empty() {
99                cfg.get_token_endpoint()
100            } else {
101                self.token_endpoint
102            },
103            credential: cfg.credential().or(self.credential),
104            authenticator: self.authenticator,
105            extra_headers,
106            extra_oauth_params: if !cfg.extra_oauth_params().is_empty() {
107                cfg.extra_oauth_params()
108            } else {
109                self.extra_oauth_params
110            },
111        })
112    }

114    /// This API is testing only to assert the token.
115    #[cfg(test)]
116    pub(crate) async fn token(&self) -> Option<String> {
117        let mut req = self
118            .request(Method::GET, &self.token_endpoint)
119            .build()
120            .unwrap();
121        self.authenticate(&mut req).await.ok();
122        self.token.lock().await.clone()
123    }
124
125    async fn exchange_credential_for_token(&self) -> Result<String> {
126        // Credential must exist here.
127        let (client_id, client_secret) = self.credential.as_ref().ok_or_else(|| {
128            Error::new(
129                ErrorKind::DataInvalid,
130                "Credential must be provided for authentication",
131            )
132        })?;
133
134        let mut params = HashMap::with_capacity(4);
135        params.insert("grant_type", "client_credentials");
136        if let Some(client_id) = client_id {
137            params.insert("client_id", client_id);
138        }
139        params.insert("client_secret", client_secret);
140        params.extend(
141            self.extra_oauth_params
142                .iter()
143                .map(|(k, v)| (k.as_str(), v.as_str())),
144        );
145
146        let mut auth_req = self
147            .request(Method::POST, &self.token_endpoint)
148            .form(&params)
149            .build()?;
150        // extra headers add content-type application/json header it's necessary to override it with proper type
151        // note that form call doesn't add content-type header if already present
152        auth_req.headers_mut().insert(
153            http::header::CONTENT_TYPE,
154            http::HeaderValue::from_static("application/x-www-form-urlencoded"),
155        );
156        let auth_url = auth_req.url().clone();
157        let auth_resp = self.client.execute(auth_req).await?;
158
159        let auth_res: TokenResponse = if auth_resp.status() == StatusCode::OK {
160            let text = auth_resp
161                .bytes()
162                .await
163                .map_err(|err| err.with_url(auth_url.clone()))?;
164            Ok(serde_json::from_slice(&text).map_err(|e| {
165                Error::new(
166                    ErrorKind::Unexpected,
167                    "Failed to parse response from rest catalog server!",
168                )
169                .with_context("operation", "auth")
170                .with_context("url", auth_url.to_string())
171                .with_context("json", String::from_utf8_lossy(&text))
172                .with_source(e)
173            })?)
174        } else {
175            let code = auth_resp.status();
176            let text = auth_resp
177                .bytes()
178                .await
179                .map_err(|err| err.with_url(auth_url.clone()))?;
180            let e: ErrorResponse = serde_json::from_slice(&text).map_err(|e| {
181                Error::new(ErrorKind::Unexpected, "Received unexpected response")
182                    .with_context("code", code.to_string())
183                    .with_context("operation", "auth")
184                    .with_context("url", auth_url.to_string())
185                    .with_context("json", String::from_utf8_lossy(&text))
186                    .with_source(e)
187            })?;
188            Err(Error::from(e))
189        }?;
190        Ok(auth_res.access_token)
191    }

193    /// Set a custom token authenticator.
194    ///
195    /// When set, the authenticator will be called to get tokens instead of using
196    /// static tokens or OAuth credentials. This allows for custom token management
197    /// such as reading from files, APIs, or other custom sources.
198    pub fn with_authenticator(mut self, authenticator: Arc<dyn CustomAuthenticator>) -> Self {
199        self.authenticator = Some(authenticator);
200        self
201    }

203    /// Add bearer token to request authorization header.
204    fn set_bearer_token(req: &mut Request, token: &str, error_msg: &str) -> Result<()> {
205        req.headers_mut().insert(
206            http::header::AUTHORIZATION,
207            format!("Bearer {token}")
208                .parse()
209                .map_err(|e| Error::new(ErrorKind::DataInvalid, error_msg).with_source(e))?,
210        );
211        Ok(())
212    }
213
214    /// Invalidate the current token without generating a new one. On the next request, the client
215    /// will attempt to generate a new token.
216    pub(crate) async fn invalidate_token(&self) -> Result<()> {
217        *self.token.lock().await = None;
218        Ok(())
219    }
220
221    /// Invalidate the current token and set a new one. Generates a new token before invalidating
222    /// the current token, meaning the old token will be used until this function acquires the lock
223    /// and overwrites the token.
224    ///
225    /// If credential is invalid, or the request fails, this method will return an error and leave
226    /// the current token unchanged.
227    pub(crate) async fn regenerate_token(&self) -> Result<()> {
228        let new_token = self.exchange_credential_for_token().await?;
229        *self.token.lock().await = Some(new_token.clone());
230        Ok(())
231    }
232
233    /// Authenticates the request by adding a bearer token to the authorization header.
234    ///
235    /// This method supports four authentication modes (in order of precedence):
236    ///
237    /// 1. **Custom authenticator** - If set, use the custom CustomAuthenticator to get tokens.
238    /// 2. **Token authentication** - Use the provided static `token` directly.
239    /// 3. **OAuth authentication** - Exchange `credential` for a token, cache it, then use it.
240    /// 4. **No authentication** - Skip authentication when none of the above are available.
241    ///
242    /// When an authenticator is provided, it takes precedence over static tokens and credentials.
243    async fn authenticate(&self, req: &mut Request) -> Result<()> {
244        // Try authenticator first (highest priority)
245        if let Some(authenticator) = &self.authenticator {
246            let token = authenticator.get_token().await?;
247            Self::set_bearer_token(req, &token, "Invalid custom token")?;
248            return Ok(());
249        }
250
251        // Clone the token from lock without holding the lock for entire function.
252        let token: Option<String> = self.token.lock().await.clone();
253
254        if self.credential.is_none() && token.is_none() {
255            return Ok(());
256        }
257
258        // Either use the provided token or exchange credential for token, cache and use that
259        let token = match token {
260            Some(token) => token,
261            None => {
262                let token = self.exchange_credential_for_token().await?;
263                // Update token so that we use it for next request instead of
264                // exchanging credential for token from the server again
265                *self.token.lock().await = Some(token.clone());
266                token
267            }
268        };
269
270        Self::set_bearer_token(req, &token, "Invalid token received from catalog server!")?;
271        Ok(())
272    }
273
274    #[inline]
275    pub fn request<U: IntoUrl>(&self, method: Method, url: U) -> RequestBuilder {
276        self.client
277            .request(method, url)
278            .headers(self.extra_headers.clone())
279    }
280
281    /// Executes the given `Request` and returns a `Response`.
282    pub async fn execute(&self, mut request: Request) -> Result<Response> {
283        request.headers_mut().extend(self.extra_headers.clone());
284        Ok(self.client.execute(request).await?)
285    }
286
287    // Queries the Iceberg REST catalog after authentication with the given `Request` and
288    // returns a `Response`.
289    pub async fn query_catalog(&self, mut request: Request) -> Result<Response> {
290        self.authenticate(&mut request).await?;
291        self.execute(request).await
292    }

295/// Deserializes a catalog response into the given [`DeserializedOwned`] type.
296///
297/// Returns an error if unable to parse the response bytes.
298pub(crate) async fn deserialize_catalog_response<R: DeserializeOwned>(
299    response: Response,
300) -> Result<R> {
301    let bytes = response.bytes().await?;
302
303    serde_json::from_slice::<R>(&bytes).map_err(|e| {
304        Error::new(
305            ErrorKind::Unexpected,
306            "Failed to parse response from rest catalog server",
307        )
308        .with_context("json", String::from_utf8_lossy(&bytes))
309        .with_source(e)
310    })
311}

313/// Deserializes a unexpected catalog response into an error.
314pub(crate) async fn deserialize_unexpected_catalog_error(response: Response) -> Error {
315    let err = Error::new(
316        ErrorKind::Unexpected,
317        "Received response with unexpected status code",
318    )
319    .with_context("status", response.status().to_string())
320    .with_context("headers", format!("{:?}", response.headers()));
321
322    let bytes = match response.bytes().await {
323        Ok(bytes) => bytes,
324        Err(err) => return err.into(),
325    };
326
327    if bytes.is_empty() {
328        return err;
329    }
330    err.with_context("json", String::from_utf8_lossy(&bytes))
